43 #define STACK_PUSH(A) (STACK[SP++] = A)
44 #define STACK_POP() (STACK[--SP])

57 int compile_bf(FILE *fp) {
58  unsigned short pc = 0, jmp_pc;
59  int c;
60  while ((c = getc(fp)) != EOF && pc < PROGRAM_SIZE) {
61  switch (c) {
62  case '>': PROGRAM[pc].operator= OP_INC_DP; break;
63  case '<': PROGRAM[pc].operator= OP_DEC_DP; break;
64  case '+': PROGRAM[pc].operator= OP_INC_VAL; break;
65  case '-': PROGRAM[pc].operator= OP_DEC_VAL; break;
66  case '.': PROGRAM[pc].operator= OP_OUT; break;
67  case ',': PROGRAM[pc].operator= OP_IN; break;
68  case '[':
69  PROGRAM[pc].operator= OP_JMP_FWD;
70  if (STACK_FULL()) return FAILURE;
71  STACK_PUSH(pc);
72  break;
73  case ']':
74  if (STACK_EMPTY()) return FAILURE;
75  jmp_pc = STACK_POP();
76  PROGRAM[pc].operator= OP_JMP_BCK;
77  PROGRAM[pc].operand = jmp_pc;
78  PROGRAM[jmp_pc].operand = pc;
79  break;
80  default: pc--; break;
81  }
82  pc++;
83  }
84  if (!STACK_EMPTY() || pc == PROGRAM_SIZE) return FAILURE;
85  PROGRAM[pc].operator= OP_END;
86  return SUCCESS;
87 }

89 int execute_bf() {
90  unsigned short data[DATA_SIZE], pc = 0;
91  unsigned int ptr = DATA_SIZE;
92  while (--ptr) { data[ptr] = 0; }
93  while (PROGRAM[pc].operator!= OP_END && ptr<DATA_SIZE) {
94  switch (PROGRAM[pc].operator) {
95  case OP_INC_DP: ptr++; break;
96  case OP_DEC_DP: ptr--; break;
97  case OP_INC_VAL: data[ptr]++; break;
98  case OP_DEC_VAL: data[ptr]--; break;
99  case OP_OUT: putchar(data[ptr]); break;
100  case OP_IN: data[ptr] = (unsigned int)getchar(); break;
101  case OP_JMP_FWD:
102  if (!data[ptr]) pc = PROGRAM[pc].operand;
103  break;
104  case OP_JMP_BCK:
105  if (data[ptr]) pc = PROGRAM[pc].operand;
106  break;
107  default: return FAILURE;
108  }
109  pc++;
110  }
111  return ptr != DATA_SIZE ? SUCCESS : FAILURE;
112 }
