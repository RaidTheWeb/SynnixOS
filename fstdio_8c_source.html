107 static size_t consume_buffer(FILE *f, char *output, ssize_t len) {
108  assert(len >= 0 && len < 1000000); // bad length to consume_buffer
109  int did_unget = unget_if_needed(f, output);
110  len -= did_unget;

120 /*
121  * consume_until is a version of consume_buffer that stops when it finds char
122  * `c`. It is indended to help implement fgets correctly.
123  *
124  * FIXME: this breaks completely if there is a NUL character in the file's
125  * buffer -- strchr will never find the split token and this will wait until
126  * EOF before returning all of the buffer.
127  */
128 static size_t consume_until(FILE *f, char *output, ssize_t len, char c) {
129  assert(len >= 0 && len < 1000000); // bad length to consume_buffer
130  int did_unget = unget_if_needed(f, output);
131  len -= did_unget;
132 
133  if (did_unget && output[0] == c) {
134  // we ungot the until char
135  // gotta bail now
136  return 1;
137  }
138  len = min(len, f->buflen);
139 
140  char *after = strchr(f->buffer, c);
141  if (after != NULL) {
142  int until_c = after - f->buffer + 1;
143  len = min(len, until_c);
144  }
145 
146  memcpy(output, f->buffer, len);
147  memmove(f->buffer, f->buffer + len, BUFSIZ - len);
148  f->buflen -= len;
149  memset(f->buffer + f->buflen, 0, BUFSIZ - f->buflen);
150  return len + did_unget;
151 }
