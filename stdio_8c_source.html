104  case 2: value = (int64_t)(int16_t)raw_value; break;
105  case 4: value = (int64_t)(int32_t)raw_value; break;
106  case 8: value = raw_value;
107  }

124  }

138  }

222  buf[i] = tmp_buf[buf_ix - i - 1];
223  }

233  const char *str, char *buf) {

236  if (format.pad.len || constrain_string_len) {

251  for (size_t i = 0; i < format.pad.len; i++) {

254  }
255  } else {
256  // If the string is longer than

258  while (*str != 0) { buf[buf_ix++] = *str++; }
259  }
260  } else {
261  while (*str != 0) { buf[buf_ix++] = *str++; }

325  assert(0); // can't have h and l in printf
326  }

329  } else if (h_count == 2) {

332  assert(0); // too many hs in printf
333  }

339  }

344  } else {

346  }

348  case 'j': // intmax_t
349  case 'z': // ssize_t

354  case '+': format.print_plus = true; goto next_char;
355  case ' ':

357  goto next_char;
358  case '-': format.pad.direction = LEFT; goto next_char;
359  case '*': format.pad.len = va_arg(args, int); goto next_char;
360  case '.': constrain_string_len = true; goto next_char;
361  case '0':
362  if (format.pad.len == 0) {
363  format.pad.c = '0';
364  goto next_char;
365  } else {
366  APPEND_DIGIT(format.pad.len, 0);
367  goto next_char;
368  }
369  case '1':
370  case '2':
371  case '3':
372  case '4':
373  case '5':
374  case '6':
375  case '7':
376  case '8':
377  case '9':
378  APPEND_DIGIT(format.pad.len, fmt[i] - '0');
379  goto next_char;
380

382  case 'd':
383  case 'i':
384  format.is_signed = true;
385  do_print_int = true;
386  break;

395  break;
396  case 'o':
397  do_print_int = true;
398  format.format = OCTAL;
399  break;
400  case 'b':
401  do_print_int = true;
402  format.format = BINARY;
403  break;
404  case 'p':
405  do_print_int = true;
406  format.format = POINTER;
407  format.bytes = sizeof(void *);
408  format.alternate_format = true;
409  format.pad.len = sizeof(void *) * 2 + 2;
410  format.pad.c = '0';
411  break;
412  case 'c':

414  buf[buf_ix++] = value;
415  break;
416  case 's':

428  buf_ix += format_float(buf + buf_ix, fvalue, format);
429  break;
430 #endif // ifndef __kernel__

432  default:;
433  // report_error
434  }
435 
436  if (do_print_int) {
437  if (format.bytes == 8) {
438  value = va_arg(args, uint64_t);
439  } else {
440  value = va_arg(args, uintptr_t);
441  }
442  buf_ix += format_int(&buf[buf_ix], value, format);
443  }
444

447  }
448  }
449

452  return buf_ix;
