172 void *heap_malloc(struct mheap *heap, size_t len) {
173  struct free_mregion *bestfit = NULL;
174  bool found_any = false;
175  assert(heap->is_init);
176  mutex_await(&heap->lock);
177 
178  if (DEBUGGING) assert_consistency(&heap->free_list);
179 
180  list_for_each(struct free_mregion, fmr, &heap->free_list, free_node) {
181  if (fmr->m.length >= len) {
182  if (!found_any || fmr->m.length < bestfit->m.length) {
183  bestfit = fmr;
184  }
185  found_any = true;
186  }
187  }
188 
189  if (!found_any) {
190  heap_expand(heap, round_up(len + sizeof(mregion), 16 * 1024 * 1024));
191  mutex_unlock(&heap->lock);
192  return heap_malloc(heap, len);
193  }
194 
195 
196  struct free_mregion *after = mregion_split(bestfit, len);
197  if (after) list_prepend(&bestfit->free_node, &after->free_node);
198 
199  list_remove(&bestfit->free_node);
200  struct mregion *mr = &bestfit->m;
201 
202  void *ptr = mregion_ptr((struct mregion *)bestfit);
203  memset(ptr, ALLOC_POISON, mr->length);
204  heap->allocations++;
205 
206  heap->free_size -= len;
207  if (after) heap->free_size -= sizeof(mregion);
208 
209  if (heap->free_size < 64 * 1024) heap_expand(heap, HEAP_BASE_LEN);
210 
211  mutex_unlock(&heap->lock);
212  return ptr;
213 }

215 void heap_free(struct mheap *heap, void *allocation) {
216  if (!allocation) return;
217  mutex_await(&heap->lock);
218  struct mregion *mr = mregion_of(allocation);
219  if (!mregion_validate(mr)) {
220  error_printf("invalid free of %p\n", allocation);
221  return;
222  }
223 
224  if (DEBUGGING) assert_consistency(&heap->free_list);
225 
226  size_t allocation_len = mr->length;
227 
228  memset(allocation, FREE_POISON, mr->length);
229  struct free_mregion *fmr = (struct free_mregion *)mr;
230  struct free_mregion *before = NULL;
231 
232  // Keep the free list sorted topologically
233  list_for_each(struct free_mregion, fl, &heap->free_list, free_node) {
234  if (fl > fmr) {
235  break;
236  } else {
237  before = fl;
238  }
239  }
240 
241  heap->frees++;

243  heap->free_size += sizeof(mregion);
244  } else if (before) {
245  list_prepend(&before->free_node, &fmr->free_node);
246  } else {
247  list_prepend(&heap->free_list, &fmr->free_node);
248  }
249 
250  heap->free_size += allocation_len;
251  mutex_unlock(&heap->lock);
252 }
