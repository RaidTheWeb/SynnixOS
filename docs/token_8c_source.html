93 bool tokenize(const char *string, list_head *out) {
94  const char *cursor = string;
95  const char *begin;
96  struct token *t;
97 
98  while (*cursor) {
99  t = NULL;
100  if (isspace(*cursor)) skip_whitespace(&cursor);
101 
102  for (int i = 0; i < ARRAY_LEN(token_info); i++) {
103  if (!token_info[i].is_simple) continue;
104  const char *tv = token_info[i].value;
105  size_t tv_len = strlen(tv);
106  if (strncmp(cursor, tv, tv_len) == 0) {
107  t = make_token(string, cursor, cursor + tv_len,
108  token_info[i].type);
109  cursor += tv_len;
110  goto next;
111  }
112  }
113 
114  switch (*cursor) {
115  case '"': // FALLTHROUGH
116  case '\'':
117  begin = cursor;
118  string_end(&cursor);
119  t = make_token(string, begin + 1, cursor - 1, TOKEN_STRING);
120  break;
121  case '$':
122  cursor++; // '$'
123  begin = cursor;
124  ident_end(&cursor);
125  t = make_token(string, begin, cursor, TOKEN_VAR);
126  break;
127  case '#':
128  // discard comment
129  line_end(&cursor);
130  break;
131  default:
132  if (isident(*cursor)) {
133  begin = cursor;
134  ident_end(&cursor);
135  t = make_token(string, begin, cursor, TOKEN_STRING);
136  } else {
137  fprintf(stderr, "Unexpected '%c' at position %zi\n", *cursor,
138  cursor - string);
139  fprintf(stderr, " > %s\n", string);
140  fprintf(stderr, " ");
141  for (int i = 0; i < cursor - string; i++) {
142  fprintf(stderr, " ");
143  }
144  fprintf(stderr, "^\n");
145  return false;
146  }
147  }
148  next:

150  }
